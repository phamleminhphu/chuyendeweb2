Hướng dẫn tải về 1 File

Đầu tiên vào https://script.google.com tạo project và past đoạn code đọc danh sách file:

function doGet(e){
  var output = HtmlService.createHtmlOutput();
  var id = e.parameters.id;
  var str = "GG Drive API<br>";
  if(id == null){
    str = str + 'Not found ID File';
  }else{
    str = str + downloadfile(id);
  }
  output.setContent(str);
  return output;
}

function downloadfile(id){
  var file = DriveApp.getFileById(id);
  file.setSharing(DriveApp.Access.ANYONE, DriveApp.Permission.VIEW);
  var str = 'Name: ' + file.getName()+': ';
  str = str + '<a href="' + file.getUrl() + '" target="_blank">View File</a> | ';
  str = str + '<a href="https://drive.google.com/uc?id=' + id + '&export=download" target="_blank">Link Download</a>';
  return str;
}
